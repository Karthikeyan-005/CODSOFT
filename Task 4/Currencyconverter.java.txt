import java.io.*;

import java.util.Scanner;

public class Currencyconverter {

    public static void main(String[] args) {

        Scanner scanner = new Scanner(System.in);


        System.out.println("Choose the base currency:");

        System.out.println("1. USD");

        System.out.println("2. EUR");

        System.out.println("3. GBP");

        System.out.println("4. JPY");

        System.out.println("5. INR");


        int baseCurrencyChoice = scanner.nextInt();

        double baseAmount;


        switch (baseCurrencyChoice) {

            case 1:

                System.out.print("Enter amount in USD: ");

                baseAmount = scanner.nextDouble();

                break;

            case 2:

                System.out.print("Enter amount in EUR: ");

                baseAmount = scanner.nextDouble();

                break;

            case 3:

                System.out.print("Enter amount in GBP: ");

                baseAmount = scanner.nextDouble();

                break;

            case 4:

                System.out.print("Enter amount in JPY: ");

                baseAmount = scanner.nextDouble();

                break;

            case 5:

                System.out.print("Enter amount in INR: ");

                baseAmount = scanner.nextDouble();

                break;

            default:

                System.out.println("Invalid choice");

                scanner.close();

                return;
        }

 
       System.out.println("Choose the target currency:");

        System.out.println("1. EUR");

        System.out.println("2. GBP");

        System.out.println("3. JPY");

        System.out.println("4. INR");


        int targetCurrencyChoice = scanner.nextInt();


        double exchangeRate = 0.0;

        String targetCurrency = "";


        switch (targetCurrencyChoice) {

            case 1:

                exchangeRate = getExchangeRate(baseCurrencyChoice, 2);

                targetCurrency = "EUR";

                break;

            case 2:

                exchangeRate = getExchangeRate(baseCurrencyChoice, 3);

                targetCurrency = "GBP";

                break;

            case 3:

                exchangeRate = getExchangeRate(baseCurrencyChoice, 4);

                targetCurrency = "JPY";

                break;

            case 4:

                exchangeRate = getExchangeRate(baseCurrencyChoice, 5);

                targetCurrency = "INR";

                break;

            default:

                System.out.println("Invalid choice");

                scanner.close();

                return;
        }

 
       double convertedAmount = baseAmount * exchangeRate;

        System.out.println("Converted amount in " + targetCurrency + ": " + convertedAmount);


        scanner.close();

    }

    private static double getExchangeRate(int baseCurrency, int targetCurrency) {

               double[][] exchangeRates = {
 
                    {1.0, 0.85, 0.75, 110.0, 75.0},
                 {1.18, 1.0, 0.88, 128.0, 88.0}, 
                {1.33, 1.13, 1.0, 146.0, 100.0}, 

                {0.0091, 0.0078, 0.0068, 1.0, 0.68},
                 {0.013, 0.011, 0.0096, 1.48, 1.0}  
       };

        return exchangeRates[baseCurrency - 1][targetCurrency - 1];

    }

}
